<VirtualHost *:443>
  ServerName sync.inceptionnotes.com

  RewriteEngine On
  RewriteCond %{HTTP:Upgrade} =websocket [NC]
  RewriteRule /(.*)           wss://localhost:8443/$1 [P,L]
  RewriteCond %{HTTP:Upgrade} !=websocket [NC]
  RewriteRule /(.*)           https://localhost:8443/$1 [P,L]

  ProxyPassReverse / http://localhost:8443/
</VirtualHost>
